The correct perspective is:

1. Condition on a carry/mode class or on an sr=60 collision class.

In other words:
> a critical pair is a **minimal correction set** for the kernel-conditioned schedule bridge.

### Practical consequence
The number of critical pairs tracks the richness of the kernel-conditioned manifold:
- more sr=60 collisions

- more ways to repair the sr=61 inconsistency by freeing bits

def count_suffix(chunk_idx, state):
    key = (chunk_idx, state.carry1, state.carry2, state.lin_id)
    if key in memo:
        return memo[key]

    if chunk_idx == NUM_CHUNKS:
        memo[key] = 1 if final_accepts(state) else 0
        return memo[key]

    succ = enumerate_chunk_transitions(chunk_idx, state)
    total = 0
    for s2 in succ:
        total += count_suffix(chunk_idx + 1, s2)

    memo[key] = total
    return total


def enumerate_chunk_transitions(chunk_idx, state):
    ctx = init_local_context(state, chunk_idx)
    out = []

    def recurse_bit(bit_in_chunk):
        if bit_in_chunk == CHUNK_BITS:
            s2 = project_to_boundary_state(ctx)
            if s2 is not None:
                out.append(s2)
            return

        # fetch affine forms for needed register bits at this bit
        a,b,c,d,e,f,g,h = get_forms(ctx, bit_in_chunk)

        # Ch selector: q = e
        for q in feasible_values(ctx, e):
            mark0 = ctx.mark()
            if not ctx.add_eq(e, q):
                ctx.rollback(mark0)
                continue
            ch = f if q else g

            # Maj selector: p = b xor c
            for p in [0,1]:
                mark1 = ctx.mark()
                if not ctx.add_xor_eq(b, c, p):
                    ctx.rollback(mark1)
                    continue
                maj = a if p else c

                recurse_adders(bit_in_chunk, ch, maj)
                ctx.rollback(mark1)

            ctx.rollback(mark0)

    def recurse_adders(bit_in_chunk, ch, maj, add_idx=0):
        if add_idx == NUM_ACTIVE_ADDERS:
            if apply_local_equalities(ctx, bit_in_chunk, ch, maj):
                recurse_bit(bit_in_chunk + 1)
            return

        spec = build_adder_spec(ctx, bit_in_chunk, add_idx, ch, maj)
        x, y, cin1, cin2 = spec.x, spec.y, spec.cin1, spec.cin2

        # 3-valued adder mode
        for mode in ["00", "01", "11"]:
            mark = ctx.mark()
            ok = True

            if mode == "00":
                ok &= ctx.add_eq(x, 0)
                ok &= ctx.add_eq(y, 0)
                set_sum_and_carry(ctx, spec, mode, cin1, cin2)

            elif mode == "01":
                ok &= ctx.add_xor_eq(x, y, 1)
                set_sum_and_carry(ctx, spec, mode, cin1, cin2)

            else:  # "11"
                ok &= ctx.add_eq(x, 1)
                ok &= ctx.add_eq(y, 1)
                set_sum_and_carry(ctx, spec, mode, cin1, cin2)

            if ok and ctx.propagate():
                recurse_adders(bit_in_chunk, ch, maj, add_idx + 1)

            ctx.rollback(mark)

    recurse_bit(0)
    return dedup(out)
